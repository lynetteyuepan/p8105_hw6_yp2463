  filter(grade %in% c("A", "B", "C")) %>% 
  mutate(inspection_num = row_number(),
         boro = str_to_title(boro)) %>% 
  filter(boro !="Missing") %>% 
  select(inspection_num, boro, grade, score, critical_flag, dba, cuisine_description,   zipcode, violation_description)
